Flexslider + Flexloader

        
<div class="flexslider">
  <ul class="slides">
    <li>
      <img src="img/1x1.gif" data-original="img/slideshow/1080/grey01.jpg">
    </li>
    ...
    <li>
      <img src="img/1x1.gif" data-original="img/slideshow/1080/grey16.jpg">
    </li>
  </ul>
</div>
        
      
        
$('#flexloader-demo-2-slideshow').flexslider({
  namespace: "flexloader-demo-2-",
  animation: "slide",
  prevText: "prev",
  nextText: "next",
  keyboard: false,
  slideshowSpeed: 100000,
  animationSpeed: 300,
  start: function(slider) {
    GLOBAL.flexslider_lazyloader.load_adjacent_slides(slider);
  },
  after: function(slider) {
    GLOBAL.flexslider_lazyloader.load_adjacent_slides(slider);
  }
});
